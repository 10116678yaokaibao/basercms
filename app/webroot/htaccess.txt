#
# サーバーによっては RewriteBase の行頭の # を削除する必要があります
# サブディレクトリに設置する場合、/ の部分を設置した場所にあわせて編集してください
# （例）サブディレクトリ basercms に設置する場合
#      RewriteBase    /basercms/
#
# ■ 動作確認済レンタルサーバー
# http://basercms.net/manuals/introductions/2.html
#

<IfModule mod_rewrite.c>
    RewriteEngine  on
#   RewriteBase    /
    RewriteCond    %{REQUEST_FILENAME} !-d
    RewriteCond    %{REQUEST_FILENAME} !-f
    RewriteRule    ^(.*)$ index.php?url=$1 [QSA,L]
</IfModule>
